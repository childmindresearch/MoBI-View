MoBI_View.views.main_app_view

Module providing the MainAppView class for MoBI_View.

Implements the main application window, which orchestrates an EEG tab (eeg_plot_widget.EEGPlotWidget), a numeric-data tab (numeric_plot_widget.MultiStreamNumericContainer), and a QtWidgets.QTreeWidget-based control panel for toggling channel visibility. Also adds a reset button in the status bar to restore the control panel.

124    def add_tree_item(self, stream_name: str, channel_name: str) -> None:
125        """Adds hierarchical items to the control panel tree for visibility control.
126
127        This method manages the creation and organization of items in the tree widget
128        that controls channel visibility. It ensures each stream appears only once as
129        a top-level item, and each channel appears as a child of its parent stream.
130
131        When a new stream is encountered, this method creates a checkable tree item
132        for it and initializes tracking structures. When a new channel is encountered,
133        it creates a child item under the appropriate stream with the display name
134        extracted from the full channel identifier. (Lazy initialization)
135
136        All items are initially created in the checked (visible) state and are made
137        user-checkable to allow toggling visibility through the control panel.
138
139        Args:
140            stream_name: Name of the LSL stream (e.g., "EEGStream").
141            channel_name: Fully qualified identifier including stream prefix
142                (e.g., "EEGStream:Fz").
143        """
144        if stream_name not in self._stream_items:
145            stream_item = QtWidgets.QTreeWidgetItem(self._tree_widget)
146            stream_item.setText(0, stream_name)
147            flags = stream_item.flags() | QtCore.Qt.ItemFlag.ItemIsUserCheckable
148            stream_item.setFlags(flags)
149            stream_item.setCheckState(0, QtCore.Qt.CheckState.Checked)
150            self._stream_items[stream_name] = stream_item
151
152        if channel_name not in self._channel_items:
153            parent_item = self._stream_items[stream_name]
154            channel_item = QtWidgets.QTreeWidgetItem(parent_item)
155            channel_item.setText(0, channel_name.split(":", 1)[-1])
156            flags = channel_item.flags() | QtCore.Qt.ItemFlag.ItemIsUserCheckable
157            channel_item.setFlags(flags)
158            channel_item.setCheckState(0, QtCore.Qt.CheckState.Checked)
159            self._channel_items[channel_name] = channel_item

161    def update_plot(self, data: dict) -> None:
162        """Updates the plots with new data from a stream.
163
164        Routes incoming data samples to the appropriate visualization tab based on
165        the stream type. EEG data is sent to the EEG tab, while all other data types
166        are sent to the numeric data tab.
167
168        Args:
169            data: A dictionary containing stream data with keys:
170                "stream_name": Name of the data stream.
171                "data": List of sample values.
172                "channel_labels": List of channel names corresponding to samples
173                    (not fully qualified).
174        """
175        stream_name = data.get("stream_name", "")
176        sample_list = data.get("data", [])
177        channel_labels = data.get("channel_labels", [])
178        for idx, val in enumerate(sample_list):
179            label = (
180                channel_labels[idx]
181                if idx < len(channel_labels)
182                else f"Channel{idx + 1}"
183            )
184            chan_name = f"{stream_name}:{label}"
185            visible = self._channel_visibility[chan_name]
186            if self._stream_types.get(stream_name) == "EEG":
187                self._eeg_tab.update_data(chan_name, val, visible)
188            else:
189                self._numeric_tab.update_numeric_containers(
190                    stream_name, chan_name, val, visible
191                )

193    def set_plot_channel_visibility(self, channel_name: str, visible: bool) -> None:
194        """Toggles the visibility of a channel.
195
196        Args:
197            channel_name: Fully qualified channel identifier (e.g., "EEGStream:Fz").
198            visible: True to show the channel, False to hide it.
199        """
200        self._channel_visibility[channel_name] = visible

210        status_bar.showMessage(f"Status: {message}")

212    def _on_tree_item_changed(self, item: QtWidgets.QTreeWidgetItem) -> None:
213        """Handles changes in the control panel tree to update channel visibility.
214
215        When a top-level (stream) item is clicked, this method applies the same
216        visibility state to all child (channel) items. When a child item is
217        clicked, it updates only that specific channel's visibility.
218
219        Args:
220            item: The tree widget item that was changed.
221        """
222        parent_item = item.parent()
223        if parent_item is None:
224            new_state = item.checkState(0)
225            for i in range(item.childCount()):
226                child = item.child(i)
227                if child is not None:
228                    child.setCheckState(0, new_state)
229        else:
230            short_name = item.text(0)
231            stream_name = parent_item.text(0)
232            full_name = f"{stream_name}:{short_name}"
233            is_visible = item.checkState(0) == QtCore.Qt.CheckState.Checked
234            self.set_plot_channel_visibility(full_name, is_visible)
